#include "libft.h"

char    *ft_substr(const char *s, unsigned int start, size_t len)
{
        char            *sub_s;
        size_t          i;

        if (len >= ft_strlen(s))
                len = ft_strlen(s) - start;
        sub_s = malloc(sizeof(char) * (len + 1));
        if (sub_s == NULL)
                return (NULL);
        if (start >= ft_strlen(s))
                return ("");
        i = 0;
        while (i < len)
        {
                sub_s[i] = (char)s[start];
                i++;
                start++;
        }
        sub_s[i] = '\0';
        return (sub_s);
}
